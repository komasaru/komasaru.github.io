    Ruby - PrawnгҒ§TweetгғӘгӮ№гғҲPDFдҪңжҲҗпјҒ

    Updated: 2011-08-15 00:20


    е°‘гҒ—еүҚгҒ«гҖҢRuby - PrawnгҒ§PDFдҪңжҲҗпјҒгҖҚгҒ§е…¬й–ӢгҒ—гҒҹгҒЁгҒҠгӮҠгҖҒRubyгҒ§гҖҢPwarnгҖҚGemгғ‘гғғгӮұгғјгӮёгӮ’дҪҝз”ЁгҒ—гҒҰPDFгӮ’дҪңжҲҗгҒҷгӮӢж–№жі•гӮ’зҙ№д»ӢгҒ—гҒҫгҒ—гҒҹгҖӮ

    д»ҠеӣһгҒҜPDFдҪңжҲҗеҮҰзҗҶгӮ’е®ҹз”ЁеҢ–гҒ—гҒҰгҒҝгҒҫгҒ—гҒҹгҖӮ

    зӢ¬еӯҰгҒ§зҝ’еҫ—гҒ—гҒҹгӮӮгҒ®гҒӘгҒ®гҒ§гҖҒгҒӮгҒҫгӮҠз¶әйә—гҒ§гҒҜгҒӮгӮҠгҒҫгҒӣгӮ“гҒҢгӮҪгғјгӮ№(LinuxзүҲ)гӮ’е…¬й–ӢгҒ—гҒҫгҒҷгҖӮ еҸӮиҖғгҒ«гҒӘгӮҢгҒ°е№ёгҒ„гҒ§гҒҷгҖӮ

    еүҚжҸҗжқЎд»¶

    • LinuxпјҲеҪ“ж–№гҒҜScientificLinux6.0пјүгӮ’дҪҝз”ЁгҒ—гҒҰгҒ„гӮӢгҖӮ
    • LinuxгҒ«RubyпјҲеҪ“ж–№гҒҜRuby1.9.2-p290пјүгҒҢгӮӨгғігӮ№гғҲгғјгғ«гҒ•гӮҢгҒҰгҒ„гӮӢгҖӮ
    • LinuxгҒ«MySQLгҒҢгӮӨгғігӮ№гғҲгғјгғ«гҒ•гӮҢгҒҰгҒ„гҒҰгҖҒдҪҝз”ЁгҒҷгӮӢгғҶгғјгғ–гғ«гҒҢдҪңжҲҗгҒ•гӮҢгҒҰгҒ„гӮӢгҖӮ
    • Ruby Gemгғ‘гғғгӮұгғјгӮёгҖҢmysql2гҖҚгҒҢгӮӨгғігӮ№гғҲгғјгғ«гҒ•гӮҢгҒҰгҒ„гӮӢгҖӮ
    • Ruby Gemгғ‘гғғгӮұгғјгӮёгҖҢsanitizeгҖҚгҒҢгӮӨгғігӮ№гғҲгғјгғ«гҒ•гӮҢгҒҰгҒ„гӮӢгҖӮ пјҲйҒҺеҺ»иЁҳдәӢгҖҢRuby - Sanitize Gemгғ‘гғғгӮұгғјгӮёгҒ®гӮӨгғігӮ№гғҲгғјгғ«пјҒгҖҚгӮӮеҸӮз…§пјү
    • Ruby Gemгғ‘гғғгӮұгғјгӮёгҖҢprawnгҖҚгҒҢгӮӨгғігӮ№гғҲгғјгғ«гҒ•гӮҢгҒҰгҒ„гӮӢгҖӮ

    RubyгӮ№гӮҜгғӘгғ—гғҲ

    MySQLгҒ«дҝқеӯҳгҒ—гҒҰгҒ„гӮӢиҮӘеҲҶгҒ®TwitterгҒ®гғ„гӮӨгғјгғҲгӮ’дёҖиҰ§гҒ«гҒҷгӮӢPDFдҪңжҲҗгӮ№гӮҜгғӘгғ—гғҲгҒ§гҒҷгҖӮ WindowsзүҲгҒ§гӮӮгҒ»гҒјеҗҢгҒҳгҒ§гҒҷгҒҢгҖҒMySQLгҒ®Gemгғ‘гғғгӮұгғјгӮёгҒҜз•°гҒӘгӮҠгҒҫгҒҷгҒ—гӮігғјгғҮгӮЈгғігӮ°гӮӮиӢҘе№Із•°гҒӘгӮҠгҒҫгҒҷгҖӮ

    • гҖҢIPA PжҳҺжңқгҖҚгҒЁгҒ„гҒҶгғ•гӮ©гғігғҲгҒҢеҝ…иҰҒгҒ§гҒҷгҖӮ
    • RubyгӮ№гӮҜгғӘгғ—гғҲгҖҒгғ•гӮ©гғігғҲгғ•гӮЎгӮӨгғ«гҒ®й…ҚзҪ®е ҙжүҖгҒҜвҖқ/home/srcвҖқгғҮгӮЈгғ¬гӮҜгғҲгғӘй…ҚдёӢгҒ§гҒҷгҖӮ
    • еј•ж•°гӮ’жҢҮе®ҡгҒӣгҒҡгҒ«е®ҹиЎҢгҒҷгӮӢгҒЁгҖҒеүҚжңҲпј‘ж—ҘгҒӢгӮүеүҚжңҲжң«ж—ҘгҒ®1гғ¶жңҲеҲҶгӮ’еҸ–еҫ—гҒ—гҒҫгҒҷгҖӮ
    • еј•ж•°гӮ’пј’гҒӨпјҲ YYYYMMDD, YYYYMMMDD пјүжҢҮе®ҡгҒҷгӮӢгҒЁз¬¬пј‘еј•ж•°гҒӢгӮүз¬¬пј’еј•ж•°гҒҫгҒ§гҒ®жңҹй–“гҒ®гғҮгғјгӮҝгӮ’еҸ–еҫ—гҒ—гҒҫгҒҷгҖӮ
    • еҮәжқҘдёҠгҒҢгӮӢPDFгҒ®гғ•гӮЎгӮӨгғ«еҗҚгҒҜвҖқTweetList_YYYYMMDD-YYYYMMDD.pdfвҖқгҒ§дҝқеӯҳе…ҲгҒҜвҖқ/home/src/twitterвҖқгҒ§гҒҷгҖӮ

    гғ•гӮЎгӮӨгғ«еҗҚпјҡtwitter_make_pdf_tweet.rb

    # -*- coding: utf-8 -*-
#-------------------------------------
# еј•ж•° :  з„ЎгҒ—
#           or
#         з¬¬пј‘пјҡ99999999(еҸ–еҫ—й–Ӣе§Ӣе№ҙжңҲж—Ҙ(YYYYMMDD))
#         з¬¬пј’пјҡ99999999(еҸ–еҫ—зөӮдәҶе№ҙжңҲж—Ҙ(YYYYMMDD))
#-------------------------------------

require 'mysql2'
require 'date'
require 'sanitize'
require 'kconv'
require 'prawn'

class TwitterMakePdfTweet

  # DB(MySQL)жҺҘз¶ҡжғ…е ұ
  $DB       = 'hoge'
  $HOST     = '127.0.0.1'
  $USER     = 'hoge'
  $PASSWORD = 'hogehoge'

  # [CLASS] еј•ж•°
  class Arg

    # INITIALIZE
    def initialize()

      # еј•ж•°еҸ–еҫ—
      @date_f = ARGV[0].to_s
      @date_t = ARGV[1].to_s

    end

    # еј•ж•°гғҒгӮ§гғғгӮҜ
    def get_date()

      begin

        # еј•ж•°гҒҢз„ЎгҒ„е ҙеҗҲ
        #   еҸ–еҫ—й–Ӣе§Ӣе№ҙжңҲж—ҘпјҡеүҚжңҲпј‘ж—Ҙ
        #   еҸ–еҫ—зөӮдәҶе№ҙжңҲж—ҘпјҡеүҚжңҲжң«ж—Ҙ
        if @date_f == "" && @date_t == ""

          # д»Ҡж—Ҙж—Ҙд»ҳеҸ–еҫ—
          date_today = Date.today

          # д»ҠжңҲпј‘ж—ҘеҸ–еҫ—
          date_this_month_first = Date.new(date_today.year, date_today.month, 1)

          # еүҚжңҲпј‘ж—ҘеҸ–еҫ—
          date_last_month_first = date_this_month_first << 1
          date_f                = date_last_month_first.strftime( "%Y%m%d" )

          # еүҚжңҲжң«ж—ҘеҸ–еҫ—
          date_last_month_last  = date_this_month_first - 1
          date_t                = date_last_month_last.strftime( "%Y%m%d" )

        # еј•ж•°гҒҢгҒӮгӮӢе ҙеҗҲ
        #   з¬¬1еј•ж•°[еҸ–еҫ—й–Ӣе§Ӣе№ҙжңҲж—Ҙ]гҒ®еҰҘеҪ“жҖ§гғҒгӮ§гғғгӮҜ
        #   з¬¬2еј•ж•°[еҸ–еҫ—зөӮдәҶе№ҙжңҲж—Ҙ]гҒ®еӯҳеңЁгғҒгӮ§гғғгӮҜ
        #   з¬¬2еј•ж•°[еҸ–еҫ—зөӮдәҶе№ҙжңҲж—Ҙ]гҒ®еҰҘеҪ“жҖ§гғҒгӮ§гғғгӮҜ
        #   (з¬¬3еј•ж•°д»ҘйҷҚгҒҜз„ЎиҰ–)
        else

          # з¬¬1еј•ж•°[еҸ–еҫ—й–Ӣе§Ӣе№ҙжңҲж—Ҙ]гҒҢ8жЎҒгҒ®ж•°еӯ—гҒӢгғҒгӮ§гғғгӮҜ
          if !( @date_f.to_s =~ /^\d{8}$/ )
            
            # гӮЁгғ©гғјгғЎгғғгӮ»гғјгӮё
            str_msg = "еј•ж•°[0]гӮЁгғ©гғј[#{@date_f.to_s}] : пјҳжЎҒгҒ®ж•°еӯ—гӮ’жҢҮе®ҡгҒ—гҒҰдёӢгҒ•гҒ„гҖӮ"
            STDERR.puts( str_msg )
            exit 1
          
          end

          # з¬¬1еј•ж•°[еҸ–еҫ—й–Ӣе§Ӣе№ҙжңҲж—Ҙ]гҒҢеҰҘеҪ“гҒӘж—Ҙд»ҳгҒӢгғҒгӮ§гғғгӮҜ
          y = @date_f[0,4].to_i
          m = @date_f[4,2].to_i
          d = @date_f[6,2].to_i
          if !Date.valid_date?( y, m, d ) # еҰҘеҪ“гҒӘж—Ҙд»ҳгҒӘгӮүtrueгҖҒгҒқгӮҢд»ҘеӨ–гҒҜnilгҒҢиҝ”гҒЈгҒҰгҒҸгӮӢ

            # гӮЁгғ©гғјгғЎгғғгӮ»гғјгӮё
            str_msg = "еј•ж•°[0]гӮЁгғ©гғј[#{@date_f.to_s}] : пјҳжЎҒгҒ§еҰҘеҪ“гҒӘж—Ҙд»ҳгӮ’жҢҮе®ҡгҒ—гҒҰдёӢгҒ•гҒ„гҖӮ"
            STDERR.puts( str_msg )
            exit 1

          end

          date_f = @date_f

          # з¬¬2еј•ж•°[еҸ–еҫ—зөӮдәҶе№ҙжңҲж—Ҙ]гҒ®еӯҳеңЁгғҒгӮ§гғғгӮҜ
          # з¬¬2еј•ж•°[еҸ–еҫ—зөӮдәҶе№ҙжңҲж—Ҙ]гҒҢ8жЎҒгҒ®ж•°еӯ—гҒӢгғҒгӮ§гғғгӮҜ
          if @date_t == "" || !( @date_t.to_s =~ /^\d{8}$/ )

            # гӮЁгғ©гғјгғЎгғғгӮ»гғјгӮё
            str_msg =  "еј•ж•°[0] : [#{@date_f.to_s}] OK!\n"
            str_msg << "еј•ж•°[1]гӮЁгғ©гғј[#{@date_t.to_s}] : пјҳжЎҒгҒ®ж•°еӯ—гӮ’жҢҮе®ҡгҒ—гҒҰдёӢгҒ•гҒ„гҖӮ"
            STDERR.puts( str_msg )
            exit 1

          end

          # з¬¬2еј•ж•°[еҸ–еҫ—зөӮдәҶе№ҙжңҲж—Ҙ]гҒҢеҰҘеҪ“гҒӘж—Ҙд»ҳгҒӢгғҒгӮ§гғғгӮҜ
          y = @date_t[0,4].to_i
          m = @date_t[4,2].to_i
          d = @date_t[6,2].to_i
          if !Date.valid_date?( y, m, d ) # еҰҘеҪ“гҒӘж—Ҙд»ҳгҒӘгӮүtrueгҖҒгҒқгӮҢд»ҘеӨ–гҒҜnilгҒҢиҝ”гҒЈгҒҰгҒҸгӮӢ

            # гӮЁгғ©гғјгғЎгғғгӮ»гғјгӮё
            str_msg =  "еј•ж•°[0] : [#{@date_f.to_s}] OK!\n"
            str_msg << "еј•ж•°[1]гӮЁгғ©гғј[#{@date_t.to_s}] : пјҳжЎҒгҒ§еҰҘеҪ“гҒӘж—Ҙд»ҳгӮ’жҢҮе®ҡгҒ—гҒҰдёӢгҒ•гҒ„гҖӮ"
            STDERR.puts( str_msg )
            exit 1

          end

          date_t = @date_t

        end

        # еӨ§е°ҸгғҒгӮ§гғғгӮҜ
        #   з¬¬1еј•ж•°[еҸ–еҫ—й–Ӣе§Ӣе№ҙжңҲж—Ҙ] > з¬¬2еј•ж•°[еҸ–еҫ—зөӮдәҶе№ҙжңҲж—Ҙ] гҒҜгӮЁгғ©гғј
        if @date_f.to_i > @date_t.to_i

            # гӮЁгғ©гғјгғЎгғғгӮ»гғјгӮё
            str_msg =  "еј•ж•°[0] : [#{@date_f.to_s}], еј•ж•°[1] : [#{@date_t.to_s}]\n"
            str_msg << "еј•ж•°[0] вүҰ еј•ж•°[1] гҒЁгҒ—гҒҰгҒҸгҒ гҒ•гҒ„гҖӮ"
            STDERR.puts( str_msg )
            exit 1

        end

        return { :date_f => date_f,
                 :date_t => date_t  }

      rescue => e

        # гӮЁгғ©гғјгғЎгғғгӮ»гғјгӮё
        str_msg = "[дҫӢеӨ–зҷәз”ҹ][" + self.class.name + ".get_date()] " + e.to_s
        STDERR.puts( str_msg )
        exit 1

      end

    end

  end

  # [CLASS] twitter_timelineгғҶгғјгғ–гғ«
  class TwitterTimeline

    # INITIALIZE
    def initialize( hash = {} )

      @date_f = hash[:date_f]
      @date_t = hash[:date_t]

    end

    # гғҮгғјгӮҝеҸ–еҫ—
    def read_data()

      begin

        # гғҮгғјгӮҝеҸ–еҫ—й–Ӣе§Ӣгғ»зөӮдәҶе№ҙжңҲж—Ҙ
        date_f = "#{@date_f[0,4]}-#{@date_f[4,2]}-#{@date_f[6,2]} 00:00:00"
        date_t = "#{@date_t[0,4]}-#{@date_t[4,2]}-#{@date_t[6,2]} 23:59:59"

        # MySQL жҺҘз¶ҡ
        my = Mysql2::Client.new( :host     => $HOST,
                                 :username => $USER,
                                 :password => $PASSWORD,
                                 :database => $DB        )

        # SQL гӮҜгӮЁгғӘе®ҹиЎҢ ( SELECT )
        sql =  "SELECT tweet_id, "
        sql << "       text, "
        sql << "       created_at, "
        sql << "       source "
        sql << "  FROM twitter_timelines "
        sql << " WHERE created_at between '" + date_f + "' AND '" + date_t + "' "
        res = my.query( sql, :as => :array )

        return res

      rescue => e

        # гӮЁгғ©гғјгғЎгғғгӮ»гғјгӮё
        str_msg = "[дҫӢеӨ–зҷәз”ҹ][" + self.class.name + ".read_date()] " + e.to_s
        STDERR.puts( str_msg )
        exit 1

      end

    end

  end

  #################
  #### гғЎгӮӨгғіеҮҰзҗҶ ####
  #################
  begin

    puts( "STARTED!!" )

    # еј•ж•°еҸ–еҫ—гғ»гғҒгӮ§гғғгӮҜ
    obj_arg = Arg.new()
    res = obj_arg.get_date()
    # иЁҲз®—еҜҫиұЎе№ҙжңҲж—Ҙ[From, To]
    date_f = res[:date_f]
    date_t = res[:date_t]
    puts( "[DATE_FROM] #{date_f}" )
    puts( "[DATE_TO  ] #{date_t}" )

    # twitter_timelineгғҶгғјгғ–гғ«
    hash = { :date_f => date_f,
             :date_t => date_t }
    tbl_twitter_timeline = TwitterTimeline.new( hash )

    # гӮҝгӮӨгғ гғ©гӮӨгғігғҮгғјгӮҝеҸ–еҫ—
    res = tbl_twitter_timeline.read_data()

    # еҸ–еҫ—гғҮгғјгӮҝпјҗд»¶гҒ®е ҙеҗҲгҒҜзөӮдәҶ
    if res.count == 0
      puts( "Data Count = 0" )
      puts( "FINISHED!")
      exit
    end

    #### [START] PDFдҪңжҲҗеҮҰзҗҶ ####

    # гғ•гӮ©гғігғҲгғ•гӮЎгӮӨгғ«иЁӯе®ҡ(IPA PжҳҺжңқ)
    FONT_FILE = 'ipamp.ttf' # Windows
    # FONT_FILE = '/home/src/ipamp.ttf' # Linux
    # дҪңжҲҗгҒҷгӮӢPDFгғ•гӮЎгӮӨгғ«еҗҚ
    PDF_FILE  = 'TweetList_' + date_f + "-" + date_t + '.pdf' # Windows
    # PDF_FILE  = '/home/src/twitter/TweetList_' + date_f + "-" + date_t + '.pdf' # Linux

    # pdfгӮӘгғ–гӮёгӮ§гӮҜгғҲгӮ’з”ҹжҲҗ
    pdf = Prawn::Document.new(
      :page_size       => "A4",      # з”ЁзҙҷгӮөгӮӨгӮә
      :page_layout     => :portrait, # з”Ёзҙҷеҗ‘гҒҚ ( зёҰ:portraitгҖҒжЁӘ:landscape )
      :top_margin      => 60,        # дҪҷзҷҪпјҲдёҠпјү
      :left_margin     => 40,        # дҪҷзҷҪпјҲе·Ұпјү
      :right_margin    => 30         # дҪҷзҷҪпјҲеҸіпјү
    )

    # гғ•гӮ©гғігғҲжҢҮе®ҡ
    pdf.font FONT_FILE

    # еҸ–еҫ—гғҮгғјгӮҝеҲҶгғ«гғјгғ—
    res.each do |row|

      tweet_time = Time.parse( row[2].to_s ).strftime( "%Y/%m/%d %H:%M:%S" ) # LinuxеҜҫеҝң
      pdf.text "#{ tweet_time}        ID[ #{row[0].to_s} ]        Source[ #{Sanitize.clean(row[3].to_s)} ]"
      pdf.pad(5) { pdf.text "#{Kconv.tosjis( row[1].to_s )}" }

      # дёӢгҒё5pxз§»еӢ•
      pdf.move_down 5

      # зҪ«з·ҡ
      pdf.stroke_horizontal_rule

      # дёӢгҒё5pxз§»еӢ•
      pdf.move_down 5

    end

    # гғҡгғјгӮёгғҳгғғгғҖ
    pdf.page_count.times do |i|

      pdf.go_to_page( i + 1 )

      # гғҡгғјгӮёгӮҝгӮӨгғҲгғ«
      pdf.bounding_box( [ 0, 780 ], :width => 215, :height => 15 ) do
        pdf.text "TweetдёҖиҰ§ [ #{ date_f.to_s } - #{ date_t.to_s } ]",
          :align => :left
        pdf.stroke do
          pdf.line pdf.bounds.bottom_left, pdf.bounds.bottom_right
        end
      end

      # гғҡгғјгӮёз•ӘеҸ·
      pdf.number_pages "<page> / <total>",
        :at          => [pdf.bounds.right - 50, 780],
        :width       => 50,
        :align       => :right,
        :page_filter => :all

    end

    pdf.render_file( PDF_FILE )

    #### [E N D] PDFдҪңжҲҗеҮҰзҗҶ ####

    puts( "FINISHED!")

  rescue => e

    # гӮЁгғ©гғјгғЎгғғгӮ»гғјгӮё
    str_msg = "[дҫӢеӨ–зҷәз”ҹ]" + e.to_s
    STDERR.puts( str_msg )
    exit 1

  end

end
